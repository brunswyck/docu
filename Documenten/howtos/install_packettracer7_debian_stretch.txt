However don't just install one package with dpkg -i
Better to put the backports repository in our system with -1 priority

root@vdab10:/lib# vi /etc/apt/sources.list.d/jessie-backports.list

root@vdab10:/lib# vi /etc/apt/preferences.d/jessie-backports.pref

root@vdab10:/lib# cat /etc/apt/sources.list.d/jessie-backports.list /etc/apt/preferences.d/jessie-backports.pref 
deb http://ftp.be.debian.org/debian jessie-backports main

Package: *
Pin: release a=jessie-backports
Pin-Priority: -1
root@vdab10:/lib# 


root@vdab10:/lib# apt-get update

root@vdab10:/lib# apt-get install -t jessie-backports libssl1.0.0
